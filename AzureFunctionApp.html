import { AzureFunctionApp, AzureFunctionAppProps } from '@gradientedge/cdk-utils'

class CustomConstruct extends AzureFunctionApp {
  constructor(id: string, props: AzureFunctionAppProps) {
    super(id, props)
    this.props = props
    this.id = id
    this.initResources()
  }

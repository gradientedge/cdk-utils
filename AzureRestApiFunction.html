import { AzureRestApiFunction, AzureRestApiFunctionProps } from '@gradientedge/cdk-utils'

class CustomConstruct extends AzureRestApiFunction {
  constructor(id: string, props: AzureRestApiFunctionProps) {
    super(id, props)

    this.id = id
    this.initResources()
  }
